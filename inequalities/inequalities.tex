\section*{Arithmetic-Geometric Mean Inequality}

For $a_1,a_2,\ldots,a_n\geq 0$,
\[
    \frac{a_1+a_2+\cdots+a_n}{n}
    \geq
    \sqrt[n]{a_1a_2\cdots a_n}.
\]
Equality iff
\[
    a_1=a_2=\cdots=a_n.
\]

\section*{Weighted AM--GM Inequality}

For $a_1,a_2,\ldots,a_n\geq 0$ and weights $\lambda_1,\lambda_2,\ldots,\lambda_n>0$,
\[
    \frac{\lambda_1a_1+\lambda_2a_2+\cdots+\lambda_na_n}{\lambda_1+\lambda_2+\cdots+\lambda_n}
    \geq
    \left(a_1^{\lambda_1}a_2^{\lambda_2}\cdots a_n^{\lambda_n}\right)^{1/(\lambda_1+\lambda_2+\cdots+\lambda_n)}.
\]
Equality iff
\[
    a_1=a_2=\cdots=a_n.
\]

\section*{Power Mean Inequality}

For $a_1,a_2,\ldots,a_n>0$ and $k_1>k_2$,
\[
    \left(\frac{1}{n}\sum_{i=1}^n a_i^{k_1}\right)^{1/k_1}
    \geq
    \left(\frac{1}{n}\sum_{i=1}^n a_i^{k_2}\right)^{1/k_2}.
\]
Equality iff
\[
    a_1=a_2=\cdots=a_n.
\]
\section*{Cauchy-Schwarz Inequality}

For real $a_i,b_i$,
\[
    \left(\sum_{i=1}^n a_ib_i\right)^2
    \leq
    \left(\sum_{i=1}^n a_i^2\right)
    \left(\sum_{i=1}^n b_i^2\right).
\]
Equality iff one vector is zero or
\[
    a_i=\lambda b_i \quad (i=1,2,\ldots,n)
\]
for some constant $\lambda$.

\section*{Titu's Lemma}

For real $a_i$ and $b_i>0$,
\[
    \frac{a_1^2}{b_1}+\frac{a_2^2}{b_2}+\cdots+\frac{a_n^2}{b_n}
    \geq
    \frac{(a_1+a_2+\cdots+a_n)^2}{b_1+b_2+\cdots+b_n}.
\]
Equality iff
\[
    \frac{a_1}{b_1}=\frac{a_2}{b_2}=\cdots=\frac{a_n}{b_n}.
\]

\section*{Chebyshev's Inequality}

If
\[
    \begin{aligned}
        a_1&\leq a_2\leq \cdots\leq a_n,\\
        b_1&\leq b_2\leq \cdots\leq b_n,
    \end{aligned}
\]
then
\[
    \frac{1}{n}\sum_{i=1}^n a_ib_i
    \geq
    \left(\frac{1}{n}\sum_{i=1}^n a_i\right)
    \left(\frac{1}{n}\sum_{i=1}^n b_i\right).
\]
If the orders are opposite, the inequality reverses. Equality iff one sequence is constant.

\section*{Rearrangement Inequality}

If
\[
    a_1\leq a_2\leq \cdots\leq a_n,
    \qquad
    b_1\leq b_2\leq \cdots\leq b_n,
\]
then for any permutation $\pi$,
\[
    \sum_{i=1}^n a_ib_{n+1-i}
    \leq
    \sum_{i=1}^n a_ib_{\pi(i)}
    \leq
    \sum_{i=1}^n a_ib_i.
\]
If both sequences are strictly increasing, right equality iff $\pi(i)=i$; left equality iff $\pi(i)=n+1-i$.

\section*{Holder's Inequality}

For $p,q>1$ with $\frac{1}{p}+\frac{1}{q}=1$,
\[
    \sum_{i=1}^n |a_ib_i|
    \leq
    \left(\sum_{i=1}^n |a_i|^p\right)^{1/p}
    \left(\sum_{i=1}^n |b_i|^q\right)^{1/q}.
\]
Equality iff one vector is zero or
\[
    |a_i|^p=\lambda |b_i|^q \quad (i=1,2,\ldots,n)
\]
for some $\lambda\geq 0$.

\section*{Minkowski's Inequality}

For $p>1$,
\[
    \left(\sum_{i=1}^n |a_i+b_i|^p\right)^{1/p}
    \leq
    \left(\sum_{i=1}^n |a_i|^p\right)^{1/p}
    +
    \left(\sum_{i=1}^n |b_i|^p\right)^{1/p}.
\]
Equality iff one vector is zero or
\[
    a_i=\lambda b_i \quad (i=1,2,\ldots,n)
\]
for some $\lambda\geq 0$.

\section*{Jensen's Inequality}

For convex $f$, $x_i$ in its domain, $\lambda_i\geq 0$, and $\sum_{i=1}^n\lambda_i=1$,
\[
    f\left(\sum_{i=1}^n \lambda_ix_i\right)
    \leq
    \sum_{i=1}^n \lambda_if(x_i).
\]
For concave $f$, the inequality reverses. If $f$ is strictly convex, equality iff all positive-weight $x_i$ are equal.

\section*{Bernoulli's Inequality}

For $x>-1$,
\[
    (1+x)^r\geq 1+rx
    \quad (r\geq 1 \text{ or } r\leq 0),
\]
\[
    (1+x)^r\leq 1+rx
    \quad (0\leq r\leq 1).
\]
For $r\neq 0,1$, equality iff $x=0$. For $r=0$ or $r=1$, equality always holds.

\section*{Schur's Inequality}

For $x,y,z\geq 0$ and $r>0$,
\[
    x^r(x-y)(x-z)+y^r(y-z)(y-x)+z^r(z-x)(z-y)\geq 0.
\]
Equality iff
\[
    x=y=z,
\]
or, after permutation,
\[
    (x,y,z)=(t,t,0) \quad (t\geq 0).
\]
For $r=1$,
\[
    x^3+y^3+z^3+3xyz
    \geq
    x^2y+x^2z+y^2x+y^2z+z^2x+z^2y.
\]
\section*{Muirhead's Inequality}

For $x_1,x_2,\ldots,x_n\geq 0$, if
\[
    (a_1,a_2,\ldots,a_n)\succ (b_1,b_2,\ldots,b_n),
\]
then
\[
    \sum_{\mathrm{sym}} x_1^{a_1}x_2^{a_2}\cdots x_n^{a_n}
    \geq
    \sum_{\mathrm{sym}} x_1^{b_1}x_2^{b_2}\cdots x_n^{b_n}.
\]
For positive variables, equality iff $x_1=x_2=\cdots=x_n$, unless the exponent tuples are the same up to permutation.

\section*{Nesbitt's Inequality}

For $a,b,c>0$,
\[
    \frac{a}{b+c}+\frac{b}{c+a}+\frac{c}{a+b}\geq \frac{3}{2}.
\]
Equality iff
\[
    a=b=c.
\]

\section*{Radon's Inequality}

For $a_i\geq 0$, $b_i>0$, and $m>0$,
\[
    \sum_{i=1}^n \frac{a_i^{m+1}}{b_i^m}
    \geq
    \frac{(a_1+a_2+\cdots+a_n)^{m+1}}{(b_1+b_2+\cdots+b_n)^m}.
\]
Equality iff
\[
    \frac{a_1}{b_1}=\frac{a_2}{b_2}=\cdots=\frac{a_n}{b_n}.
\]
